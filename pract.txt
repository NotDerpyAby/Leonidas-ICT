        // Game variables
        let movingBox = document.getElementById("moving-box");

        // Move the box randomly
        function moveBox() {
            let container = document.getElementById("game-container");
            let maxX = container.clientWidth - movingBox.clientWidth;
            let maxY = container.clientHeight - movingBox.clientHeight;

            let randomX = Math.floor(Math.random() * maxX);
            let randomY = Math.floor(Math.random() * maxY);

            movingBox.style.left = randomX + "px";
            movingBox.style.top = randomY + "px";
        }

        // Click event for the game
        movingBox.addEventListener("click", moveBox);

        // Move the box every second
        setInterval(moveBox, 1000);
